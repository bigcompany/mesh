Mesh emitter websocket gateway

You can connect to this gateway using big or the engine.io client 

  var big = require('big');
  big.start({ 
    host: 'localhost',
    port: 8888 
  }, function (err){
    if (err) throw err;
    var emitter = big.resource.mesh.emitter;
    emitter.on('ping', function(data){
      console.log('hello', data);
    });
    emitter.emit('pong');
  });
